Q 01. Write a C++ program that generates a random number between 1 and 100. The program should repeatedly ask the user to guess the number and display whether the guess is too low, too high, or correct. The loop should continue until the user guesses the correct number.

Ans. #include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand(time(0));
    int number = rand() % 100 + 1;
    int guess;

    cout << "Guess the number between 1 and 100" << endl;

    do {
        cout << "Enter your guess: ";
        cin >> guess;

        if (guess < number)
            cout << "Too LOW!" << endl;
        else if (guess > number)
            cout << "Too HIGH!" << endl;
        else
            cout << "Correct! You guessed it!" << endl;

    } while (guess != number);

    return 0;
}
